22 inline std::string centerText(const std::string& text, int totalWidth) {
23  if (text.size() >= totalWidth) {
24  return text.substr(0, totalWidth); // Truncate if text is too wide
25  }
26  int paddingLeft = (totalWidth - text.size()) / 2;
27  int paddingRight = totalWidth - text.size() - paddingLeft; // Remaining padding goes to the right
28  return std::string(paddingLeft, ' ') + text + std::string(paddingRight, ' ');
29 }
